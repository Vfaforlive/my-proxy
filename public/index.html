JAVASCRIPT / NODE.JS 
// server.js — Kern des Web-Proxys
const express = require('express');
const fetch   = require('node-fetch');
const app     = express();

app.use(express.static('public'));  // Frontend ausliefern

// ① Haupt-Route: empfängt Anfrage vom Browser
app.get('/proxy', async (req, res) => {
  const target = req.query.url;          // z.B. https://example.com
  if (!target) return res.status(400).send('Keine URL');

  // ② Server fetcht die Zielseite (nicht der Browser!)
  const response = await fetch(target, {
    headers: { 'User-Agent': 'Mozilla/5.0' }
  });

  let html = await response.text();

  // ③ Alle Links umschreiben → laufen wieder durch Proxy
  html = html.replace(/href="(https?:\/\/.*?)"/g,
    (_, u) => `href="/proxy?url=${encodeURIComponent(u)}"`
  );

  // ④ Umgeschriebenes HTML an Browser senden
  res.send(html);
});

app.listen(3000, () => console.log('Proxy läuft auf :3000'));
URL-REWRITING DETAIL 
// Das Herzstück: URL-Rewriting
// Ohne das würden Links auf der Zielseite direkt laden
// und an der Firewall scheitern.

function rewriteHtml(html, baseUrl) {
  const base = new URL(baseUrl);

  // Absolute Links: href="https://..."
  html = html.replace(
    /href="(https?:\/\/[^"]+)"/g,
    (_, u) => `href="/proxy?url=${encodeURIComponent(u)}"`
  );

  // Relative Links: href="/pfad/seite"
  html = html.replace(
    /href="(\/[^"]+)"/g,
    (_, path) => `href="/proxy?url=${encodeURIComponent(base.origin + path)}"`
  );

  // Bilder umschreiben: src="https://..."
  html = html.replace(
    /src="(https?:\/\/[^"]+)"/g,
    (_, u) => `src="/proxy-asset?url=${encodeURIComponent(u)}"`
  );

  return html;
}

// Bilder & Assets durchleiten
app.get('/proxy-asset', async (req, res) => {
  const resp = await fetch(req.query.url);
  resp.body.pipe(res);  // Stream direkt weiterleiten
});

Wie es funktioniert
SCHRITT 01

Browser → Proxy

Du gibst eine URL ein. Dein Browser sendet die Anfrage an /proxy?url=... auf deinem Server — nicht direkt zur Zielseite. Die Firewall sieht nur diese eine Verbindung.

SCHRITT 02

Server fetcht serverseitig

Node.js ruft die Zielseite mit fetch() auf. Das passiert auf dem Server, nicht im Browser — der Server steht außerhalb der Firewall oder im Internet.

SCHRITT 03

URL-Rewriting

Alle Links im HTML werden umgeschrieben: href="https://x.com" wird zu href="/proxy?url=https://x.com". So läuft auch Navigation weiter durch den Proxy.

SCHRITT 04

Response zurück

Das umgeschriebene HTML wird an den Browser geliefert. Der Nutzer surft normal — jede Seite läuft unsichtbar durch den Proxy-Server.

⚠️
Hinweis: Das Umgehen von Unternehmens- oder Schul-Firewalls kann gegen Nutzungsbedingungen oder lokale Gesetze verstoßen. Dieses Projekt ist für Lernzwecke und den Einsatz auf eigenen Netzwerken gedacht. Bitte prüfe vorher, was in deinem Kontext erlaubt ist.
